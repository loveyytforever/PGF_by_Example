\documentclass{standalone}
\usepackage{pgfplots}
\pgfplotsset{compat=newest}
\usepgfplotslibrary{polar}
\begin{document}
\begin{tikzpicture}
	\begin{polaraxis}[title={Degrees and/or Radians}]
	\addplot 
		coordinates {(0,1) (90,1) (180,1) (270,1)};
	\addlegendentry{Deg}
	\addplot+[data cs=polarrad] 
		coordinates {(0,1.5) (pi/2,1.5) 
		  (pi,1.5) (pi*3/2,1.5)};
	\addlegendentry{Rad}
	\end{polaraxis}
\end{tikzpicture}
\end{document}
